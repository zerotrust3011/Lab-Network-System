Site 1 (dhcp, vlantrunking, định tuyến ,subinter)

ena	
conf t
host Site1
int g0/1 
no shut
ex
int g0/1.10
en dot 10
ip add 192.168.10.1 255.255.255.0
ex
int g0/1.20
en dot 20
ip add 192.168.20.1 255.255.255.0
ex
ip dhcp ex 192.168.10.1 192.168.10.10
ip dhcp ex 192.168.20.1 192.168.20.10
ip dhcp pool  vlan10
net 192.168.10.0 255.255.255.0
def 192.168.10.1
dns 8.8.8.8
ex
ip dhcp pool  vlan20
net 192.168.20.0 255.255.255.0
def 192.168.20.1
dns 8.8.8.8
ex
int g0/2
no shut
ip add 192.168.50.1 255.255.255.0
ex
int g0/0
no shut 
ip add 192.168.70.1 255.255.255.0
ex
route eigrp 10
no au
net 192.168.20.0 
net 192.168.10.0 
net 192.168.50.0 
net 192.168.70.0 
ex
